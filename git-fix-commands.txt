# Fix Git Repository - Comandi corretti per Replit

# Il repository esiste già, non serve git init
# Usa questi comandi nella shell Replit:

# 1. Verifica stato attuale
cd /home/runner/workspace
pwd
ls -la
git remote -v
git status

# 2. Se il remote non è configurato, aggiungilo:
git remote add origin https://github.com/AScozzari/EasyCashFlows.git

# 3. Se il remote esiste già ma è sbagliato, correggilo:
git remote set-url origin https://github.com/AScozzari/EasyCashFlows.git

# 4. Verifica branch
git branch

# 5. Se non sei su main, cambia:
git checkout main

# 6. Aggiungi tutti i file e committa
git add .
git commit -m "Complete project deployment $(date +%Y%m%d_%H%M%S)"

# 7. Push al repository
git push -u origin main

# Se ricevi errore "non-fast-forward", forza il push:
# git push -f origin main

echo "=== Git configurato correttamente ==="
echo "Repository: https://github.com/AScozzari/EasyCashFlows.git"
echo "Branch: main"
